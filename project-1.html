class Config():
    def __init__( self ) :
        self.config = {}
        self.file   = 'properties.txt'
        global path
        self.path = path
        with open( self.path + self.file ) as file:
            for line in file:
                tup = line.split( '=' )
                key = tup[  0 ].strip()
                val = tup[ -1 ].strip()
                self.config[ key ] = val
    def getdict( self ) :
        return self.config
