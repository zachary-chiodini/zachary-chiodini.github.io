def popup( msg, title, length = 100 ) :
    ''' Generates a popup with title and message '''
    def organize( msg ) :
        '''
        Creates next line 
        characters where necessary 
        '''
        lst = []
        while True :
            if len( msg ) <= length :
                lst.append( msg )
                break
            slc = msg[ 0 : length ]
            idx = slc[ ::-1 ].index( ' ' )
            idx = length - idx
            lst.append( msg[ 0 : idx ] )
            msg = msg[ idx : ]
        return '\n'.join( lst )
    pop = tk.Tk()
    pop.wm_title( title )
    msg = organize( msg )
    lbl = ttk.Label( pop, text = msg )
    lbl.pack( padx = length, pady = msg.count( '\n' ) )
    btn = ttk.Button( pop, text = 'Okay', 
                      command = pop.destroy )
    btn.pack()

class Runtime():
    def __init__( self ):
        self.start = time.time()
        self.time  = 0
        self.stop  = 0
        self.days  = 0
        self.hours = 0
        self.mins  = 0
        self.secs  = 0
        self.ave   = 0
        self.total = 0
        self.cycle = 1
        self.log   = ' '
        self.prog  = '' # Progress of bar
        self.path  = '' # Current directory

    def print( self, s ):
        print( s, end = '\r', flush = True )
            
    def clear( self ):
        print( '\r', end = '' )

    def log_file( self, log = None, end = '\n' ):
        with open( self.path + 'log.txt', 'a',
                   encoding = 'utf-8', errors = 'ignore' ) as file:
            if log: s = log
            else  : s = self.log
            file.write( s + end )

    def jotdown( self, s, flush = True, logfile = True ):
        if flush:
            if self.prog :
                if self.log == self.prog :
                    self.clear()
                    self.log = s
                    self.print( self.prog )
                else :
                    self.clear()
                    self.log = s
                    self.print( self.prog + ' ' + self.log )
            else :
                self.clear()
                self.log = s
                self.print( self.log )
            if logfile :
                self.log_file()
        else : print( s )

    def endpage( self ):
        log = '#' * 139
        self.log_file( log )
            
    def countdown( self, s, n, logfile = True ):
        for i in range( n, -1, -1 ):
            self.jotdown( s.format( str( i ) ) )
            time.sleep( 1 )
    
    def convert( self, time ):
        self.days  = int( time / 86400 )
        self.hours = int( time % 86400 / 3600 )
        self.mins  = int( time % 3600 / 60 )
        self.secs  = int( time % 60 )
    
    def gettime( self, convert = True ):
        self.stop  = time.time()
        self.time  = self.stop - self.start
        self.start = time.time()
        if convert: self.convert( self.time )
        
    def logtime( self, flush = True, get = True ):
        if get : self.gettime()
        p = 'Runtime: {:02d}:{:02d}:{:02d}:{:02d}.'
        p = p.format( self.days, self.hours, self.mins, self.secs )
        if flush : self.jotdown( p )
        else : print( p )
        
    def progbar( self, flush = True ):
        '''
        Recurrence relation:
        Ave_( n ) = [ Ave_( n - 1 ) * ( n - 1 ) / n ] + [ x_( n ) / n ]
        '''
        n = self.cycle
        self.cycle += 1
        self.gettime( convert = False )
        self.ave = self.ave * ( n - 1 ) / n + ( self.time / n )
        self.convert( self.ave * ( self.total - n ) )
        per = ( n / self.total ) * 100
        inr = int( per )
        bar = '#' * inr
        bar = bar.ljust( 100, '-' )
        if inr == 100:
            p = 'Time Remaining: 00:00:00:00. Progress: 100% {}'.format( bar )
            self.prog = p
            self.jotdown( self.log, flush = flush, logfile = False )
        else: 
            p = 'Time Remaining: {:02d}:{:02d}:{:02d}:{:02d}. Progress: {:5.2f}% {}'
            p = p.format( self.days, self.hours, self.mins, self.secs, per, bar )
            self.prog = p
            self.jotdown( self.log, logfile = False )

    def reset( self ):
        self.start = time.time()
        self.time  = 0
        self.stop  = 0
        self.days  = 0
        self.hours = 0
        self.mins  = 0
        self.secs  = 0
        self.ave   = 0
        self.total = 0
        self.cycle = 1

    def clear_log( self ):
        with open( 'log.txt', 'w+' ) as file:
            file.truncate()

class Tools():
    def __init__( self ):
        self.log = ''

        self.track = Runtime()

    def remove_dupes( self, l ) :
        r = []
        for val in l :
            if val not in r :
                r.append( val )
        return r

    def extract_between( self, s, start, stop, strip = '', inclusive = False ):
        '''
        Returns a list of substrings that exist within a string s 
        and sandwiched between substring start and substring stop.
        '''
        l = []
        j, k = len( start ) - 1, len( stop ) - 1
        while True:
            if start in s and stop in s:
                m = s.index( start )
                n = s.index( stop )
                n += len( stop ) if inclusive else 0
            else:
                return l
            if m > n:
                s = s[ m : ]
            else:
                l.append( s[ m + j + 1 : n ].strip( strip ) )
                s = s[ n + k : ]

    def clean_accents( self, s ):
        accents = {
            'á' : 'a', 'à' : 'a', 'ã' : 'a', 'ă' : 'a', 'â' : 'a', 
            'é' : 'e', 'è' : 'e', 'ê' : 'e', 'í' : 'i', 'ì' : 'i', 'ĩ' : 'i',
            'ó' : 'o', 'ò' : 'o', 'õ' : 'o', 'ô' : 'o', 'ơ' : 'o',
            'ú' : 'u', 'ù' : 'u', 'ũ' : 'u', 'ư' : 'u', 'ü' : 'u',
            'ý' : 'y', 'ỳ' : 'y', 'ć' : 'c', 'ç' : 'c', ' du ': ' of '
            }
        for key in accents.keys():
            if key in s:
                val = accents[ key ]
                s   = s.replace( key, val )
        return s

    def remove_accents( self, df ):
        cols = [ col for col in df.columns if 'NM' in col ]
        df[ cols ] = df[ cols ].apply( lambda series : series.apply( self.clean_accents ) )
        return df

    def clean_rate( self, nm ) :
        name = []
        if '(' in nm and ')' in nm:
            for para in self.extract_between( nm, '(', ')' ):
                nm = nm.replace( para, '' )
            nm = nm.replace( '(', '' ).replace( ')', '' )
        if '-' in nm:
            nm = nm.split( '-' )[ 0 ]
        if '/' in nm:
            for s in nm.split( '/' ):
                s = s.strip()
                if len( s ) > 2:
                    name.append( s )
        if name == [] :
            name.append( nm )
        return name

    def valid_program( self, rate, archive ):
        # Check master all rates file for rate program code
        where  = archive[ 'rate_program' ] == rate
        select = archive[ where ]
        if select.empty:
            self.handoff = None
            return False
        # Get the rate program name
        names = select[ 'rate_name' ]
        names = names.str.lower().unique().tolist()
        for nm in names :
            nm = self.remove_entity_type( nm.lower() )
            for n in nm :
                if n not in names :
                    names.append( n )
        clean = []
        for nm in names :
            nm = self.clean_rate( nm )
            for n in nm :
                if n not in clean and n != '' :
                    clean.append( n )
        if clean == [] :
            self.handoff = None
            return False
        self.log = 'Found {} program names : {}'
        self.track.jotdown( self.log.format( len( clean ),
                            ', '.join( clean ) ) )
        self.handoff = clean
        return True

    def acnt_in_web( self, account, hierarchy, spaces = True ):
        if spaces: match = lambda string : account in string or string in account
        else     : match = ( lambda string : account in string.replace( ' ', '' )
                             or string.replace( ' ', '' ) in account )
        cols = [ col for col in hierarchy.columns if 'NM' in col ]
        search = hierarchy[ cols ].apply( lambda series : series.str.lower().apply( match ) )
        self.handoff = hierarchy[ search.any( axis = 1 ) ]
        if self.handoff.empty :
            # fuzzy match at 90% to correct for mispellings
            if spaces :
                match = lambda series : series.apply( fuzz.QRatio, args = [ account ] )
            else :
                match = lambda series : series.str.replace( ' ', '' ).apply(
                    fuzz.QRatio, args = [ account ]
                    )
            search = hierarchy[ cols ].apply( match ) >= 90
            self.handoff = hierarchy[ search.any( axis = 1 ) ]
        if self.handoff.empty :
            p = 'Account {} not found in hierarchy.'
        else :
            p = 'Account {} found in hierarchy.'
        self.track.jotdown( p.format( account ) )
        return not self.handoff.empty

    def duns_in_web( self, duns, hierarchy ):
        cols = [ col for col in hierarchy.columns if 'DUNS' in col ]
        find = hierarchy[ cols ] == duns
        return find.any( axis = 1 ).any( axis = 0 )

    def reverse( self, s ):

    def clean_nm( self, name ):

    def lowest_level_deduped( self, account, hierarchy, spaces = True ):
        cols = [ c for c in hierarchy.columns if 'NM' in c ]
        nms  = hierarchy[ cols ]
        # Clean name and hierarchy so that they will match
        acc = self.clean_nm( account )
        nms = nms.apply( lambda series : series.apply( self.clean_nm ) )
        # (1) substring match
        if spaces: match = lambda string : acc in string or string in acc
        else     : match = ( lambda string : acc in string.replace( ' ', '' )
                             or string.replace( ' ', '' ) in acc )
        snip = nms.apply( lambda series : series.str.lower().apply( match ) )
        rows = snip.any( axis = 1 )
        cols = snip.any( axis = 0 )
        nms  = nms.loc[ rows, cols ]
        # (2) fuzzy match
        if spaces: match = lambda string : fuzz.QRatio( string, acc )
        else     : match = lambda string : fuzz.QRatio( string.replace( ' ', '' ), acc )
        vals = nms.apply( lambda series : series.apply( match ) )
        maxx = max( vals.max( axis = 1 ) )
        mtrx = vals == maxx
        priority_queue = {
            'ULTIMATE_PARENT_ACCOUNT_NM' : 0, 'IMMEDIATE_PARENT_ACCOUNT_NM' : 1,
            'IMMEDIATE_CHILD_ACCOUNT_NM' : 2, 'CHILD_ACCOUNT_NM'            : 3
            }
        for col in self.reverse( list( priority_queue.keys() ) ):
            rows = mtrx[ col ]
            if rows.any( axis = 0 ):
                colm = col
                break
        # Lowest level best-matched accout name as it appears in AW
        name = hierarchy.loc[ rows, colm ].iloc[ 0 ]
        # Dedupe (select highest level account with the same name)
        rows = ( hierarchy == name ).any( axis = 1 )
        cols = ( hierarchy == name ).any( axis = 0 )
        slct = min( priority_queue[ col ]
                    for col in hierarchy.loc[ rows, cols ].columns )
        colm = list( priority_queue.keys() )[ slct ]
        cols = [ colm.replace( '_ACCOUNT_NM',  '_URI' ), colm,
                 colm.replace( '_ACCOUNT_NM', '_DUNS' ) ]
        rslt = hierarchy.loc[ rows, cols ]
        indx = ( rslt == name ).any( axis = 1 )
        rslt = rslt.loc[ indx, cols ].iloc[ 0 ]
        for col in rslt.index :
            if 'DUNS' in col :
                colm = col
                break
        duns = rslt[ colm ]
        for col in rslt.index :
            if '_NM' in col :
                colm = col
                break
        acnt = rslt[ colm ]
        if not duns : duns = 'not found'
        p = 'Linked to account {} with D-U-N-S {} in AW Sales Hierarchy.'
        self.track.jotdown( p.format( acnt, duns ) )
        return rslt
        
    def email_domain( self, address ):

    def separate_duns( self, record ):
        ''' Separates DUNS number from account name, when applicable. '''
        if pd.isna( record ): return '', ''
        duns = record[ -10 : -1 ]
        return ( record[ 0 : -12 ], duns ) if duns.isdigit() else ( record, '' )
        
    def scrub_duns( self, df, substr = 'ACCOUNT_NM', newstr = 'DUNS' ):
        ''' 
        Separates DUNS numbers from all account name columns 
        and appends new columns containing them. 
        '''
        cols = [ col for col in df.columns if substr in col ]
        duns = [ col.replace( substr, newstr ) for col in cols ]
        take = df.columns.tolist()
        rslt = pd.DataFrame()
        for col, duns in zip( cols, duns ):
            scrb = df[ col ].apply(
                lambda x : pd.Series( self.separate_duns( x ), index = [ col, duns ] )
                )
            rslt = pd.concat( [ rslt, scrb ], axis = 1 )
            take.remove( col )
        return pd.concat( [ df[ take ], rslt ], axis = 1 )

    def remove_entity_type( self, s ) :
        entity_types = [
            ' inc',  ' llp', ' co',  ' gpp', ' corp',
            ' llc',  ' ltd', ' hb', ' gmbh',  ' gds',
            ' non gds', ' corporation', ' incorporated',
            ' limited', ' company', ' nongds', ' lp',
            ' & co'   , ' oy', ' and co', ' preferred',
            ' group'  , 'companies', ' of companies',
            ' group of companies' ]
        sliced = [ s ]
        for suffix in entity_types :
            if suffix in s :
                # Suffix should exist at the end of the string
                if suffix.index( suffix ) + len( suffix ) == len( s ) :
                    n = s.replace( suffix, '' ).strip()
                    if n not in sliced :
                        sliced.append( n )
        return sliced

class Query():
    def __init__( self ): 
        self.Select  = ''
        self.From    = ''
        self.LJoin   = ''
        self.RJoin   = ''
        self.On      = ''
        self.Where   = ''
        self.And     = ''
        self.Or      = ''
        self.Groupby = ''
        self.Having  = ''
        self.Orderby = ''
        self.Limit   = ''
        self.handoff = ''
        self.log     = ''
        self.track   = Runtime() # track queries
        self.tools   =   Tools()

        global path
        self.path = path

    def execute( self ):
        SELECT  = (  'SELECT '     + self.Select  ) * bool( self.Select  )
        FROM    = ( ' FROM '       + self.From    ) * bool( self.From    )
        LJOIN   = ( ' LEFT JOIN '  + self.LJoin   ) * bool( self.LJoin   )
        RJOIN   = ( ' RIGHT JOIN ' + self.RJoin   ) * bool( self.RJoin   )
        ON      = ( ' ON '         + self.On      ) * bool( self.On      )
        WHERE   = ( ' WHERE '      + self.Where   ) * bool( self.Where   )
        AND     = ( ' AND '        + self.And     ) * bool( self.And     )
        OR      = ( ' OR '         + self.Or      ) * bool( self.Or      )
        GROUPBY = ( ' GROUP BY '   + self.Groupby ) * bool( self.Groupby )
        HAVING  = ( ' HAVING '     + self.Having  ) * bool( self.Having  )
        ORDERBY = ( ' ORDER BY '   + self.Orderby ) * bool( self.Orderby )
        LIMIT   = ( ' LIMIT '      + self.Limit   ) * bool( self.Limit   )
        request  = SELECT + FROM +   LJOIN + RJOIN  + ON      + WHERE
        request += AND    + OR   + GROUPBY + HAVING + ORDERBY + LIMIT
        return request

    def attempt( self, query, conxn, req = False, n = 0, i = 1 ):
        ''' Attempts the Query and Cleans the Data '''
        while n < 3:
            try:
                df = pd.read_sql( query, conxn, coerce_float = False ).astype( str )
                rows, cols = len( df ), len( df.columns )
                if 'ibm_db_dbi' in str( conxn ):
                    p = 'Cleaning D-U-N-S from {} rows and {} columns ...'.format( rows, cols )
                    self.track.jotdown( p )
                    df = self.tools.scrub_duns( df )
                self.track.jotdown( 'Returned {} rows and {} columns.'.format( rows, cols ) )
                self.track.logtime( flush = True )
                return self.tools.remove_accents( df )
            except Exception as e:
                self.track.jotdown( str( e ).replace( '\n', ' ' ).capitalize() )
                if req : raise Exception( e )
                time.sleep( 3 ) # Retry after 3 seconds
            n += 1
        if i == 2: raise Exception( 'Query terminally failed. Process terminated.' )
        conxn.close()
        s = 'Query failed after 3 attempts. Sleeping for {} seconds ...'
        self.track.countdown( s, 10 )
        self.track.jotdown( 'Attempting to reconnect ...' )
        if 'ibm_db_dbi' in str( conxn ): conxn = Connect().aw()
        else: conxn = Connect().netezza() 
        return self.attempt( query, conxn, i = 2 )

    def request( self, sql = None, conxn = None ):
        global connection
        if  connection: conxn = connection
        if sql is None: sql = self.query.execute()
        self.track.reset()
        self.track.jotdown( 'Requesting query: {}'.format( sql ) )
        return self.attempt( sql, conxn, req = True )

class Verify():
    def __init__( self ):
        # Cache
        self.valid_domains   = []
        self.invalid_domains = []
        self.valid_emails    = []
        self.invalid_emails  = []
        self.log = ''
        self.res = ''
        self.isp = ''
        self.track = Runtime()

    def call_backup( self ):
        query = Query()
        self.isp = query.isp_free_domains() + query.isp_2008()
        self.res = query.res_history()
        
    def append_cache( self, kind, address, valid ):
        if kind == 'email':
            if valid: self.valid_emails.append( address )
            else    : self.invalid_emails.append( address )
        elif kind == 'domain':
            if valid: self.valid_domains.append( address )
            else    : self.invalid_domains.append( address )
        else: raise Exception( 'Is the address an email or domain?' )

    def email_syntax( self, address ):
        ''' Checks email syntax. '''
        if '@' in address:
            split = address.split( '@' )
            try: split.remove( '' ) # incase of "@@" entries
            except: pass
            if len( split ) == 2:
                if '.' in split[ -1 ]: return True
        return False
    
    def valid_domain( self, address, timeout = 160 ):
        ''' 
        Retrieves the MX records for a given domain. 
        If records exist, the domain is likely valid.
        '''
        track = self.track
        if address in self.valid_domains:
            self.log = 'Found in Cache.'
            track.jotdown( self.log )
            return True
        if address in self.invalid_domains:
            self.log = 'Found in Cache.'
            track.jotdown( self.log )
            return False
        if address in self.isp:
            self.log = 'Found in ISP Files.'
            track.jotdown( self.log )
            self.append_cache( 'domain', address, False ) 
            return False
        if address in self.res:
            self.log = 'Found in Reservation Match History'
            track.jotdown( self.log ) 
            return True
        track.jotdown( 'Querying {} ...'.format( address ) )
        resolver = dns.resolver.Resolver()
        resolver.lifetime = resolver.timeout = timeout
        attempt  = 1
        while True:
            start = time.time()
            # Query MX records and IP address for domain
            try:
                mx = [ str( mx.exchange ) for mx in resolver.query( address, 'MX' ) ]
                track.jotdown( 'Domain {} is likely valid.'.format( address ) )
                self.log = '{} MX records: {}'.format( address, ', '.join( mx ) )
                track.jotdown( self.log )
                self.append_cache( 'domain', address, True ) 
                return True
            except Exception as e:
                stop = time.time()
                if int( stop - start ) == resolver.timeout:
                    track.jotdown( 'DNS query timed out for {}.'.format( address ) )
                    if attempt > 2: # Attempts 3 times if exception is due to timeout
                        self.log = 'Query failed for {}.'.format( address )
                        track.jotdown( self.log )
                        self.append_cache( 'domain', address, False )
                        return False
                    attempt += 1
                    track.countdown( 'Query timed out. Sleeping for {} seconds ...', 3 )
                    track.jotdown( 'Attempt {} ...'.format( attempt ) )
                    continue
                self.log = str( e ).capitalize()
                track.jotdown( self.log )
                self.append_cache( 'domain', address, False )
                return False

    def server_connect( self, server, mx, port ):
        try:
            server.connect( mx, port )
            return True
        except Exception as e:
            self.track.jotdown( str( e ).replace( '\n', ' ' ) )
            return False

    def email_connect( self, address, domain, resolver, port, use_ssl, use_tls ):
        # Query MX records for domain
        mx = [ ( mx.exchange, mx.preference ) for mx in resolver.query( domain, 'MX' ) ]
        mx = sorted( mx ) # Prioritize MX records
        i, n = 0, len( mx )
        for mxrec, pref in mx:
            i += 1
            mxrec = str( mxrec ) # argument must be str or None
            p = 'Using MX record {} of {}: {} with preference {}.'.format( i, n, mxrec, pref )
            self.track.jotdown( p )
            # SMTP setup
            if use_ssl:
                server = smtplib.SMTP_SSL( timeout = resolver.timeout )
            else: server = smtplib.SMTP( timeout = resolver.timeout )
            server.set_debuglevel( False )
            if not self.server_connect( server, mxrec, port ):
                continue
            # SMTP conversation
            if use_tls:
                server.ehlo()
                server.starttls()
                server.ehlo()
            else:
                server.helo( server.local_hostname )
            server.mail( '' )
            code, msge = server.rcpt( address )
            server.quit()
            msge = msge.decode( 'UTF-8' ).replace( '\n', ' ' )
            self.log = '{} code: {}; message: {}'.format( address, code, msge )
            self.track.jotdown( self.log )
            if code == 250: return True
        return False

    def valid_mailbox( self, address, timeout = 160, port = 25,
                       use_ssl = False, use_tls = False ):
        ''' Checks if an email mailbox exists.'''
        domain = address.split( '@' )[ -1 ]
        if address in self.valid_emails:
            self.log = 'Found in Cache.'
            self.track.jotdown( self.log )
            return True
        if address in self.invalid_emails:
            self.log = 'Found in Cache.'
            self.track.jotdown( self.log )
            return False
        if domain in self.invalid_domains:
            self.log = 'Found in Cache.'
            self.track.jotdown( self.log )
            self.append_cache( 'email', address, False )
            return False
        if domain in self.isp:
            self.log = 'Found in ISP Files.'
            self.track.jotdown( self.log )
            self.append_cache( 'email', address, False )
            return False
        self.track.jotdown( 'Querying {} ...'.format( address ) )
        resolver = dns.resolver.Resolver()
        resolver.lifetime = resolver.timeout = timeout
        attempt = 1
        while True:
            start = time.time()
            try:
                if self.email_connect( address, domain, resolver, port, use_ssl, use_tls ): 
                    self.append_cache( 'email', address, True )
                    return True
                self.append_cache( 'email', address, False )
                return False
            except Exception as e:
                stop = time.time()
                if int( stop - start ) == resolver.timeout:
                    self.track.jotdown( 'Server timed out for {}.'.format( address ) )
                    if attempt > 2: # Attempts 3 times if exception is due to timeout
                        self.log = 'Server failed for {}.'.format( address )
                        self.track.jotdown( self.log )
                        self.append_cache( 'email', address, False )
                        return False
                    attempt += 1
                    self.track.countdown( 'Query timed out. Sleeping for {} seconds ...', 3 )
                    self.track.jotdown( 'Attempt {} ...'.format( attempt ) )
                    continue
                self.log = str( e ).capitalize()
                self.track.jotdown( self.log )
                self.append_cache( 'email', address, False )
                return False

    def valid_mail( self, address, timeout = 160, port = 25 ):
        if self.email_syntax( address ):
            self.track.jotdown( 'Querying mailbox {} ...'.format( address ) )
            return self.valid_mailbox( address, timeout = timeout, port = port )
        self.track.jotdown( 'Querying domain {} ...'.format( address ) )
        return self.valid_domain( address, timeout = timeout )

    def multistep_validation( self,   address,
                              sleep   =    30, scrape  =  None,
                              timeout =   160, port    =    25, results = 100,
                              backup  = False, use_ssl = False, use_tls = False ):
        # verify.valid_domain checks
        # (1) Whether domain exists in Reservation Match History
        # (2) Whether domain exists in ISP files
        # (3) Whether domain MX records and IP address exist
        if ( self.isp == '' and self.res == '' ) and backup is True:
            self.track.jotdown( 'Opening backup files ...' )
            self.call_backup()
        domain = Tools().email_domain( address )
        if not self.valid_domain( domain, timeout ):
            return False
        if 'Reservation Match History' in self.log:
            return True
        # (4) If address is a mailbox, does it pass the SMTP?
        if self.email_syntax( address ):
            if self.valid_mailbox( address, timeout = 160, port = 25,
                                   use_ssl = False, use_tls = False ):
                return True
        # (5) Query domain as website and scrape contact emails
        if not scrape:
            scrape = Scrape()
        scrape.sleep = sleep
        emails = scrape.emails_from_website( domain )
        # (6) Do any scraped emails pass the SMTP?
        for email in emails:
            if self.valid_mailbox( email, timeout = 160, port = 25,
                                   use_ssl = False, use_tls = False ):
                return True
        # (7) Scrape all emails having the specified domain from
        #     n Google search results
        scrape.sleep = sleep
        scrape.rslts = results
        emails = scrape.email( domain )
        # (8) Do any scraped emails pass the SMTP?
        for email in emails:
            if self.valid_mailbox( email, timeout = 160, port = 25,
                                   use_ssl = False, use_tls = False ):
                return True
        self.log = 'Research could not validate.'
        return False

    def clean_html( self, s, l = '<', r = '>' ):
        while True:
            if l in s and r in s:
                i, j = s.index( l ), s.index( r ) + len( r )
                if i >= j:
                    s = s[ j : ]
                    continue
                t = s[ i : j ]
                s = s.replace( t, '' )
            elif l in s:
                s = s[ : s.index( l ) ]
            elif r in s:
                s = s[ s.index( r ) + len( r ) : ]
            else: return s

    def parse_email( self, s, start, stop, top = '' ):
        '''
        Returns a list of substrings that exist preceding a string stop and
        the first encounter of the substring start.
        '''
        special = [ 
            '!',  '#', '$', '%', '^', '&', '(', ')', '=',
            '+',  '|', '{', '}', '"', ':', ';', ',', '?',
            '/', '\\', '~', '`', "'", '[', ']', 'email-'
        ]
        l = []
        if  ' @ ' in stop: at = ' @ '
        elif '@'  in stop: at = '@'
        else: at = '(g)'
        while True:
            if stop not in s:
                return sorted( list( set( l ) ) )
            try:
                j = s.index( stop ) + len( stop )
                t = s[ : j - len( stop ) ]
                i = len( t ) - t[ : : - 1 ].index( start[ : : -1 ] )
                r =  s[ i : j ]
                r = self.clean_html( r )
                sep = r.split( at )
                usr = sep[ 0 ]
                for char in special:
                    if char in usr:
                        usr = usr.split( char )[ - 1 ]
                r = usr + '@' + sep[ -1 ]
                b1 = not usr.isdigit() and not usr.isdecimal()
                b2 = '*' not in usr and '...' not in usr and usr != ''
                if b1 and b2:
                    r += top
                    l.append( r )
                s = s[ j : ]
                if stop not in s:
                    return sorted( list( set( l ) ) )
            except ValueError:
                s = s[ j : ]

    def parse_from_start_to_stop( self, s, start, stop ):
        # for copyright
        l = []
        while True:
            if start not in s:
                return list( set( l ) )
            try:
                i = s.index( start ) + len( start )
                t = s[ i : ]
                j = len( s ) + t.index( stop ) - len( t )
                r = s[ i : j ].replace( stop, '' ).strip()
                r = self.clean_html( r )
                l.append( r )
                s = s[ i : ]
                if start not in s:
                    return list( set( l ) )
            except ValueError:
                s = s[ i : ]
        
        
    def parse_and_clean( self, s, start, stop, strip = ' ', inclusive = False, n = None ):
        clean = []
        for t in self.tools.extract_between( s, start, stop, strip, inclusive ):
            if n:
                if len( t ) > n: continue
            clean.append( self.clean_html( t ).replace( '\n', ' ' ).replace( '\xa0...', '' ) )
        return clean

    def request( self, url ):
        if self.sleep > 0:
            self.track.countdown( 'Scrape.request is sleeping for {} seconds ...', self.sleep )
        self.total += 1
        req = urllib.request.Request( url, headers = { 'User-Agent' : 'Mozilla/5.0' } )
        try: response = urllib.request.urlopen( req )
        except Exception as e: return str( e )
        code  = response.read().decode( 'UTF-8'     , 'ignore' )
        code += response.read().decode( 'ISO-8859-1', 'ignore' )
        code = code.lower()
        self.html = code
        return code

    def exception( self, engine, e ):
        log = '{} query encountered "{}" after executing {} queries in {} seconds.'
        end = int( time.time() - self.start )
        self.track.jotdown( log.format( engine, e, self.total, end ) )
        time.sleep( 3 )
        self.track.jotdown( 'Switching engines ...' )
        self.error[ engine ] = str( e )
        self.track = 0
        self.start = time.time()
        return self.search()

            if self.engine.lower() == 'bing':

            if self.engine.lower() == 'yahoo':

        if 'Google' not in self.error:

        if 'Bing'   not in self.error:
            try: return self.bing()

    def clean_titles( self, scrape, dbs = [ '' ] ):
        bad = [ 'working at', 'company profile', 'local business',
                'jobs', 'compare', 'careers', 'related searches',
                'competitors, revenue and employees', ' home',
                "'", '"' ]
        special = [ '|', ' - ', ' -', '- ', '...',   '®',
                    '–',   '—',  ';',  ':',   '»', '.,.' ]
        entity_types = [
            ' inc',  ' llp', ' co.',  ' gpp', ' corp',
            ' llc',  ' ltd', ' hb', ' gmbh',  ' gds',
            ' non gds',   ' corporation', ' incorporated',
            ' limited', ' company', ' nongds', ' lp',
            ' & co'   , ' oy', ' and co', ' preferred',
            ' group', ' companies' ]
        result = []
        # Taking linkedin, glassdoor, etc company title names
        for title in scrape:
            for site in dbs:
                if site in title:
                    title = title.replace( site, '' ).strip()
                    if '&amp;' in title:
                        title = title.replace( '&amp;', '&' )
                    for char in special:
                        if char in title:
                            title = title.split( char )[ 0 ].strip() 
                    for word in bad:
                        if word in title:
                            title = title.replace( word, '' ).strip()
                    if 'reviews' in title:
                        title = title.split( 'reviews' )[ 0 ].strip()
                    if '(' in title and ')' in title:
                        parenthesis = self.tools.extract_between( title, '(', ')' )
                        for string in parenthesis:
                            title = title.replace( string, '' ).strip()
                        title = title.replace( '(', '' ).replace( ')', '' ).strip()
                    if     ' vs ' in title : continue
                    if        title ==  '' : continue
                    if        title == ' ' : continue
                    if     title == 'home' : continue
                    if title == 'homepage' : continue
                    if    len( title ) < 4 : continue
                    if    len( title ) > 50: continue
                    if '/' in title :
                        if len( title ) - title.index( '/' ) - 1 < 3 :
                            result.append( title )
                        else :
                            for name in title.split( '/' ) :
                                result.append( name.strip() )
                        continue
                    result.append( title )
        result = self.remove_dupes( result )
        for i in range( len( result ) ):
            name = result[ i ].lower()
            name = Tools().clean_accents( name )
            result[ i ] = name
        clean = result
        for name in result:
            for suffix in entity_types:
                if suffix in name:
                    if name in result :
                        indx = result.index( name )
                        name = name.replace( suffix, '' )
            if name not in clean :
                clean.append( name )
        return self.remove_dupes( clean )

    def archived( self ) :
        '''
        Helper Function:
        Search the master alignment files for
        the analyte. If archived, return true.
        '''
        if 'mail' in self.issueid :
            rate = False
            colm = 'email_address_txt'
            find = 'DUNS'
            arch = self.mail_archive
        if 'rate' in self.issueid :
            rate = True
            colm = 'rate_program'
            find = 'account_uri'
            arch = self.rate_archive
        indx = arch[ colm ] == self.analyte
        slct = arch.loc[ indx, find ]
        if slct.empty: return False
        # Retrieve the archived links
        colms = [ 'CHILD_URI',
                  'CHILD_ACCOUNT_NM' ] 
        links = pd.DataFrame()
        for uid in slct.unique().tolist() :
            if rate :
                link = self.query.uri( uid, colms )
            else :
                link = self.query.duns( uid, colms )
            if not link.empty :
                links = links.append( link )
        # Store the archived links
        for link in links.values.tolist():
            link = ', '.join( link )
            if link not in self.info[ 'Archived Link' ] :
                self.info[ 'Archived Link' ].append( link )
        self.handoff = links
        return True

    def valid_archive( self ) :
        '''
        Helper Function:
        Validate the archived analyte linkage
        by searching the found linkage in the
        archived hierarchy.
        '''
        arch = self.handoff # from archived
        if arch.empty: # Not found in AW
            log  = 'Archived D-U-N-S or URI not found in AW'
            self.info[ 'Archived Hierarchy' ].append( log )
            self.info[ 'Archived Link'      ].append( log )
            log += ". Can't validate the archived link."
            self.track.jotdown( log )
            return False
        # Blow up hierarchy
        for indx, series in arch.iterrows() :
            uri = series[ 'CHILD_URI' ]
            web = self.query.blowup( uri, uri  = True,
                                     only_duns = True )
            # Store archived hierarchy
            ultm = self.query.ultm
            for ultm in ultm.values.tolist() :
                join = ', '.join( ultm )
                if join not in self.info[ 'Archived Hierarchy' ] :
                    self.info[ 'Archived Hierarchy' ].append( join )
            # Search the D-U-N-S number of the
            # linkage in the archived hierarchy
            if not self.linkage.empty:
                link = self.linkage
                colm = [ col for col in link.index if 'DUNS' in col ]
                duns = link[ colm ].iloc[ 0 ]
                if duns != '' :
                    if self.tools.duns_in_web( duns, web ) :
                        self.archive = series
                        # Search acnt in archived hierarchy
                        if self.tools.acnt_in_web( self.account, web ) :
                            self.acnt_in_arch = True
                        else :
                            # To account for orphans, search all the
                            # company names in the archived hierarchy
                            for name in self.info[ 'Associated Account' ] :
                                if self.tools.acnt_in_web( name, web ) :
                                    self.orphans = True
                        return True
            # Search acnt in archived hierarchy
            if self.tools.acnt_in_web( self.account, web ) :
                self.archive = series
                self.acnt_in_arch = True
                return True
            # To account for orphans, search all the
            # company names in the archived hierarchy
            for name in self.info[ 'Associated Account' ] :
                if self.tools.acnt_in_web( name, web ) :
                    self.archive = series
                    self.orphans = True
                    return True
        # If no link was found,
        # just take what's archived
        if self.linkage.empty :
            self.archive = arch.iloc[ 0 ]
            return True
        return False

    def validate_archive( self ) :
        '''
        Helper Function:
        Attempts to validate archived. If the
        archived link can't be validated, the
        archive is assumed to be valid.
        '''
        # Search archive for analyte
        if self.archived() :
            self.analyte_arch = True
            # Validate archive
            if self.valid_archive() :
                self.validarchive = True

    def has_duns( self ) :
        link = self.linkage
        for col in link.index :
            if 'DUNS' in col :
                colm = col
                break
        duns = link[ colm ]
        if duns : return True
        return False

    def find_linkage( self ) :
        '''
        Helper Function:
        If an organization was found in AW DB,
        then link analyte to the lowest level
        account. Then, provide the resolution.
        '''
        if not self.archive.empty :
            # Archive is valid
            if 'incorrect' in self.issueid :
                if self.linkage.empty :
                    self.resolve = 'Program Could Not Validate'
                    return
            self.linkage = self.archive
            if self.acnt_in_arch :
                self.resolve = 'Deny - Correctly Linked'
                if len( self.info[ 'Hierarchy' ] ) > 1 :
                    self.resolve += ' - Broken Hierarchy' 
            elif self.acnt_in_link :
                self.resolve = 'Approved'
                if len( self.info[ 'Hierarchy' ] ) > 1 :
                    self.resolve += ' - Broken Hierarchy' 
            else :
                self.resolve = 'Deny - Different Hierarchy'
        elif self.acnt_in_link :
            self.resolve = 'Approved'
            if self.entity_in_aw :
                self.handoff = self.entityresult
                self.link_analyte( self.entity_title )
                if self.parent_in_aw :
                    if not self.has_duns() :
                        self.handoff = self.parentresult
                        self.link_analyte( self.parent_title )
            elif self.parent_in_aw :
                self.handoff = self.parentresult
                self.link_analyte( self.parent_title )
            if len( self.info[ 'Hierarchy' ] ) > 1 :
                self.resolve += ' - Broken Hierarchy'
        elif self.entity_in_aw or self.parent_in_aw :
            self.resolve = 'Deny - Different Hierarchy'
            if self.entity_in_aw :
                self.handoff = self.entityresult
                self.link_analyte( self.entity_title )
            elif self.parent_in_aw :
                self.handoff = self.parentresult
                self.link_analyte( self.parent_title )
        if self.analyte_arch and not self.validarchive :
            self.resolve += ' - Delete & Relink'
        
    def link_duns( self ) :
        '''
        Helper Function:
        If linked account in AW does not have
        a D-U-N-S number, link the analyte to
        the ultimate parent account, instead.
        '''
        link = self.linkage
        for col in link.index :
            if 'DUNS' in col :
                colm = col
                break
        duns = link[ colm ]
        for col in link.index :
            if 'URI' in col :
                colm = col
                break
        auri = link[ colm ]
        if not duns :
            link = self.query.ult_acnt( auri, uri = True )
            if not link.empty :
                self.linkage = link.iloc[ 0 ]

    def scan_cache( self ) :
        '''
        First Step: Check the cache to see if
        data point has already been processed.
        '''
        if ( self.account,
             self.analyte,
             self.issueid ) in self.links.keys() :
             self.linkage = self.links[ ( self.account,
                                          self.analyte,
                                          self.issueid ) ]
             self.resolve = self.resol[ ( self.account,
                                          self.analyte,
                                          self.issueid ) ]
             # Finalize information store
             for key in self.info.keys():
                 val = self.info[ key ]
                 self.info[ key ] = ' | '.join( val )
             self.info[ 'Resolution' ] = self.resolve
             pd.set_option( 'display.max_colwidth', 10000 )
             self.info = pd.DataFrame(
                 index   = self.info.keys(),
                 data    = self.info.values(),
                 columns = [ 'Info' ] )
             # End
             self.track.clear()
             self.total.logtime()
             self.total.endpage()
             return self.resolve
        # If not found,
        return

    def validation( self ) :
        '''
        Second Step: Validate Rate Program by
        Finding it in Reltio, or Validate the
        Domain by Querying its MX Records.
        '''
        if 'mail' in self.issueid :
            if self.verify.valid_domain( self.analyte ) :
                return True
            self.resolve = 'Deny - Invalid Email Address'
            return False
        if 'rate' in self.issueid :
            if self.tools.valid_program( self.analyte,
                                    self.reltio_rates ) :
                return True
            self.resolve = 'Pending - Rate not Found in ' +\
                           'Current Source HPP Data. '    +\
                           'DSR will Revisit this Item.'
            return False
        raise ValueError(
            'The positional argument "issueid" must be ' +\
            'of type string containing "rate" or "mail."' )

    def assoc_comp( self, name = None ):
        '''
        Third Step: Search and Seize the Name
        of the Organization that's Associated
        with the Analyte.
        '''
        if self.parents :
            comps = self.scrape.parent_org( name )
        elif 'mail' in self.issueid :
            comps = self.scrape.assoc_comp( self.analyte )
            if not comps :
                comps.append( self.analyte.split( '.' )[ 0 ] )
                comps.append( self.analyte )
        elif 'rate' in self.issueid :
            comps = self.tools.handoff # From valid_program
        if len( comps ) > 5 :
            # Don't want to spend an hour here
            comps = comps[ 0 : 5 ]
        for acnt in comps :
            if fuzz.QRatio( self.account, acnt ) != 100 :
                # Correct for minor mispellings
                if fuzz.QRatio( self.account, acnt ) >= 90 :
                    comps.append( self.account )
                    break
        if self.parents :
            for name in comps :
                if name not in self.info[ 'Parent Account' ] :
                    self.info[ 'Parent Account' ].append( name )
        else :
            for name in comps :
                if name not in self.info[ 'Associated Account' ] :
                    self.info[ 'Associated Account' ].append( name )
        return comps

    def name_in_aw( self, name ) :
        '''
        Fourth Step: Query name in AW to find
        whether it exists in the database.
        '''
        # Avoid querying small substrings
        if len( name ) >  6 : suff =  True
        else                : suff = False
        if len( name ) > 14 : like =  True
        else                : like = False
        # A common error is to not include
        # spaces or include unnecessary spaces
        name = name.replace( ' ', '' )
        if self.query.acnt_in_db( name, like = like,
                                  suffix  =  suff,
                                  spaces  = False,
                                  orphans =  True ) :
            self.handoff = self.query.handoff
            cols = [ col for col
                     in self.handoff.columns
                     if 'DUNS' in col ]
            duns = self.handoff[ cols ]
            duns = duns != ''
            duns = duns.any( axis = 1 )
            if not duns.any( axis = 0 ) :
                # Accounts not at all associated with
                # a D-U-N-S number are not reliable
                return False
            self.handoff = self.handoff.loc[
                duns.index.tolist(), : ]
            if self.parents :
                self.parent_in_aw = True
                self.parentresult = self.handoff
                self.parent_title = name
            else :
                self.entity_in_aw = True
                self.entityresult = self.handoff
                self.entity_title = name   
            return True
        # Not in AW
        return False

    def link_analyte( self, name ):
        '''
        Fifth Step: Link analyte to best name
        match in AW at the highest level
        '''
        name = name.replace( ' ', '' )
        query = self.handoff
        link  = self.tools.lowest_level_deduped(
            name, query, spaces = False
            )
        self.linkage = link

    def blowup_web( self ) :
        '''
        Sixth Step: Blowup sales hierarchy of
        the linked organization name
        '''
        link = self.linkage
        # Retrieve name as it is written in AW
        for col in link.index :
            if 'NM' in col :
                colm = col
                break
        acnt = link[ colm ]
        # Store name
        if self.parents :
            if acnt not in self.info[ 'Parent Acc. in AW' ] :
                self.info[ 'Parent Acc. in AW' ].append( acnt )
        else :
            if acnt not in self.info[ 'Assoc. Acc. in AW' ]:
                self.info[ 'Assoc. Acc. in AW' ].append( acnt )
        # Blow up hierarchy
        web = self.query.blowup( acnt, only_duns = True,
                                 orphans = True )
        # Store hierarchy
        for ultm in self.query.ultm.values.tolist():
            ultm = ', '.join( ultm )
            if ultm not in self.info[ 'Hierarchy' ] :
                self.info[ 'Hierarchy' ].append( ultm )
        return web

    def acnt_in_web( self, web ) :
        '''
        Seventh Step: Search the account name
        in the sales hierarchy of linked name.
        Web is short for hierarchy.
        '''
        if web.empty : return False
        name = self.account.replace( ' ', '' )
        if self.tools.acnt_in_web( name, web,
                                   spaces = False ) :
            self.acnt_in_link = True
            return True
        return False

    def resolution( self ) :
        '''
        Last Step: Check the archive file and
        validate link, append cache with link
        resolution, "wrap up" the information
        store, and then return the resolution.
        '''
        # Check and, if link was found,
        # validate archive
        self.validate_archive()
        # Find you a link!
        self.find_linkage()
        # Store found linkage
        if self.linkage.empty :
            self.linkage = 'Program Could Not Find Link'
            self.info[ 'Found Link' ].append( self.linkage )
        else :
            self.link_duns()
            link = ', '.join( self.linkage.tolist() )
            self.info[ 'Found Link' ].append( link )
            # If link has no D-U-N-S number,
            # contact account deployment.
            for col in self.linkage.index :
                if 'DUNS' in col :
                    colm = col
                    break
            if self.linkage[ colm ] == '' :
                self.resolve += ' - Contact Account Deployment'
        if not self.resolve :
            self.resolve = 'Program Could Not Validate'
        # Append cache
        self.links[ ( self.account,
                      self.analyte,
                      self.issueid )
                  ] = self.linkage
        self.resol[ ( self.account,
                      self.analyte,
                      self.issueid )
                  ] = self.resolve
        # Finalize information store
        for key in self.info.keys():
            val = self.info[ key ]
            self.info[ key ] = ' | '.join( val )
        self.info[ 'Resolution' ] = self.resolve
        pd.set_option( 'display.max_colwidth', 10000 )
        self.info = pd.DataFrame(
            index   = self.info.keys(),
            data    = self.info.values(),
            columns = [ 'Info' ] )
        # End
        self.track.clear()
        self.total.logtime()
        self.total.endpage()
        return self.resolve

    def reset( self ) :
        '''
        Helper Function:
        Reset processing parameters.
        '''
        self.resolve = ''
        self.handoff = ''
        self.parents = False
        self.orphans = False
        self.linkage = pd.Series()
        self.archive = pd.Series()
        self.parent_title = ''
        self.entity_title = ''
        self.analyte_arch = False
        self.validarchive = False

    def processor( self, account, analyte, issueid, sleep = 30 ):
        '''
        +-------------------------------------------------+
        |             Feedback File Processor             |
        +-------------+-----------------------------------+
        |   Argument  |            Description            |
        +-------------+-----------------------------------+
        | (1) account | (1) Sales Hierarchy               |
        | (2) analyte | (2) Rate Program or Email Address |
        | (3) issueid | (3) Incorrect/Missing Rate/Email  |
        +-------------+-----------------------------------+
        '''
        # Reset timers
        self.process.track.reset()
        self.process.total.reset()
        self.process.scrape.sleep = sleep
        # Letting you know we've started
        log = 'Processing account {} with analyte {} ...'
        log = log.format( account, analyte )
        self.process.track.jotdown( log )
        # Commence program sequence
        self.process.account = account.lower()
        self.process.analyte = analyte.split( '@' )[ -1 ]
        self.process.issueid = issueid.lower()
        # Reset parameters
        self.process.reset()
        # (1) Check cache 
        cache = self.process.scan_cache()
        if cache is not None: return cache
        # (2) Is analyte valid?
        if self.process.validation() :
            # (3) Find the company that is
            #     associated with the analyte
            trash = [] # Avoid infinite loop
            prnts = [] # parent companies
            names = self.process.assoc_comp()
            while names :
                name = names.pop( 0 ) #LIFO queue
                if name in trash :
                    continue # Skip
                if name in prnts :
                    self.process.parents =  True
                else :
                    self.process.parents = False
                if len( trash ) > 10 :
                    break # Just give up
                # (4) Search company name in AW
                if self.process.name_in_aw( name ) :
                    # (5) If found, link name to analyte
                    self.process.link_analyte( name )
                    # (6) Search account in hierarchy
                    web = self.process.blowup_web()
                    if self.process.acnt_in_web( web ) :
                        # (7) If found, validate archive
                        return self.process.resolution()
                trash.append( name )
                self.process.parents = True
                pars = self.process.assoc_comp( name )
                for comp in pars :
                    if comp not in prnts :
                        prnts.append( comp )
                    if comp not in names :
                        names.append( comp )
        # Check archive
        return self.process.resolution()
